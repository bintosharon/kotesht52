##SIMPLE PIPELINE###

pipeline
{
    agent any
        stages{
            stage('Build'){
               steps{
                echo 'This is build step'
               }
            }
            stage('Test'){
               steps{
                echo 'This is Test step'
               }
            }
            stage('Deploy'){
               steps{
                echo 'This is Deploy step'
               }
            }
        }
}

}



###ENVIRONMENT VARIABLES PIPELINE#######

                }

}


#####RETRY PIPELINE#####

pipeline
{
    agent any
        stages{
            stage('Build'){
               steps{
                 retry(3){
                  sh 'This step will not work as there is no linux command in shell : so retrying'
                 }
                
               }
            }
        }
}


#### TIMEOUT PIPELINE ########

pipeline{
    agent any
    stages{
        stage('My stage'){
            steps{
                retry(3){
                    sh 'echo 3 times'
                }
                
                timeout(time :3,unit: 'SECONDS'){
                    sh 'sleep 15'
                }
            }
            
        }
    }
}
